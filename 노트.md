- 맵 ( 이중배열 ) combination

: 전 위치에서 같은 행만 전 열+1부터 시작해야하고 그 다음 행부터는 0부터 시작해야 한다. 다른 포문으로 처리해야 한다.

``` c++
void selection(int cnt,int x,int y){
    if(cnt>=3)
        return;
    if(y>=m){
      	y=0; x++;
        if(x>=n)
            return;
    }
    
    for(int j=y;j<m;j++){
        if(mymap[x][j]==0){
            mymap[x][j]=1;
            selection(cnt+1,x,j+1);
            mymap[x][j]=0;
        }
    }
  
    for(int i=x+1;i<n;i++){
        for(int j=0;j<m;j++){
            if(mymap[i][j]==0){
                mymap[i][j]=1;
                selection(cnt+1,i,j+1);
                mymap[i][j]=0;
            }
        }
    }
}
```
